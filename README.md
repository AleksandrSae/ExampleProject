# To build and run

$ git clone --recurse-submodules https://github.com/AleksandrSae/ExampleProject.git
$ cd ExampleProject/
$ git fetch origin parametrized_test:parametrized_test
$ mkdir build
$ cd build/
$ cmake ..
$ ./tst/ExampleProject_tst
